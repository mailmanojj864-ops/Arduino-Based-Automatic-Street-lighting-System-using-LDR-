// C++ code
int ldrpin = A0;
int ledpin = 9;
int ldrvalue = 0;

void setup()
{
  pinMode(ledpin, OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  ldrvalue = analogRead(ldrpin);
  Serial.println(ldrvalue);
  
  if(ldrvalue<500)
  {
    digitalWrite(ledpin,HIGH);
  }
  else
  {
    digitalWrite(ledpin,LOW);
  }
  delay(2000);
}
